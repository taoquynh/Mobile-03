func daoMotChuoi() {
    print("Nhập vào chuỗi bất kì")

    print()

func veTamGiacCan() {
    print("Nhập chiều cao tam giác, chiều cao > 1: ")
    let h = Int(readLine()!)!
    
    for i in 0..<h {
        for j in 0..<2*h{
            if j >= (h-1-i) && j <= (h-1+i){
                print("*",terminator:"")
                
            } else {
                print(" ",terminator:"")
                
            }
        }
        print()
    }
    
}
